Implementação: PID

 1     def pid(s, sp):
 2         (...)
 3 
 4         e      = sp - s           # Error
 5         e_last = sp - s_last
 6 
 7         int_e += dt*(e+e_last)/2  # Integral
 8         de_dt  = (e-e_last)/dt    # Derivative
 9 
10         p = kp * e                # PID
11         i = ki * int_e
12         d = kd * de_dt
13 
14         return p+i+d
